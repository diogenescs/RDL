select
    sum(case extract(month from l.dataPgto) when 1 then l.valorPgto else 0 end) ,
    sum(case extract(month from l.dataPgto) when 2 then l.valorPgto else 0 end) ,
    sum(case extract(month from l.dataPgto) when 3 then l.valorPgto else 0 end) ,
    sum(case extract(month from l.dataPgto) when 4 then l.valorPgto else 0 end) ,
    sum(case extract(month from l.dataPgto) when 5 then l.valorPgto else 0 end) ,
    sum(case extract(month from l.dataPgto) when 6 then l.valorPgto else 0 end) ,
    sum(case extract(month from l.dataPgto) when 7 then l.valorPgto else 0 end) ,
    sum(case extract(month from l.dataPgto) when 8 then l.valorPgto else 0 end) ,
    sum(case extract(month from l.dataPgto) when 9 then l.valorPgto else 0 end) ,
    sum(case extract(month from l.dataPgto) when 10 then l.valorPgto else 0 end),
    sum(case extract(month from l.dataPgto) when 11 then l.valorPgto else 0 end),
    sum(case extract(month from l.dataPgto) when 12 then l.valorPgto else 0 end)
from Lancamento l where extract(year from l.dataPgto) = ?1 and l.despesa = ?2 and l.ativo = 1 and l.realizado = 'S' and l.usuarioProjeto = ?3
